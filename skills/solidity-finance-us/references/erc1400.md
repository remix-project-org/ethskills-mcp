# ERC-1400 Security Token Standard — Reference

## Overview
ERC-1400 is a family of standards for security tokens. It introduces **partitioned balances**
(tranches), transfer restrictions, and controller operations needed for regulated securities.

## Key Interfaces

```solidity
// IERC1400.sol
interface IERC1400 is IERC20 {
    // Partition management
    function totalPartitions() external view returns (bytes32[] memory);
    function partitionsOf(address tokenHolder) external view returns (bytes32[] memory);
    function balanceOfByPartition(bytes32 partition, address tokenHolder) external view returns (uint256);
    function totalSupplyByPartition(bytes32 partition) external view returns (uint256);

    // Partition transfers
    function transferByPartition(bytes32 partition, address to, uint256 value, bytes calldata data)
        external returns (bytes32);
    function operatorTransferByPartition(bytes32 partition, address from, address to,
        uint256 value, bytes calldata data, bytes calldata operatorData) external returns (bytes32);

    // Operators
    function authorizeOperator(address operator) external;
    function revokeOperator(address operator) external;
    function authorizeOperatorByPartition(bytes32 partition, address operator) external;
    function revokeOperatorByPartition(bytes32 partition, address operator) external;
    function isOperator(address operator, address tokenHolder) external view returns (bool);
    function isOperatorForPartition(bytes32 partition, address operator, address tokenHolder)
        external view returns (bool);

    // Issuance / Redemption
    function issueByPartition(bytes32 partition, address tokenHolder, uint256 value, bytes calldata data) external;
    function redeemByPartition(bytes32 partition, uint256 value, bytes calldata data) external;
    function operatorRedeemByPartition(bytes32 partition, address tokenHolder, uint256 value,
        bytes calldata data, bytes calldata operatorData) external;

    // Transfer Validity
    function canTransferByPartition(bytes32 partition, address to, uint256 value, bytes calldata data)
        external view returns (bytes1, bytes32, bytes32);
    // Returns: ESC code, application-specific reason, target partition
    // ESC codes: 0x50=success, 0x51=invalid receiver, 0x52=invalid operator,
    //            0x53=insufficient balance, 0x54=insufficient allowance, 0x5F=unknown reason

    // Document management (for offering docs, prospectus etc.)
    function getDocument(bytes32 name) external view returns (string memory, bytes32);
    function setDocument(bytes32 name, string calldata uri, bytes32 documentHash) external;

    // Controller operations (forced transfers for legal/regulatory compliance)
    function isControllable() external view returns (bool);
    function controllerTransfer(address from, address to, uint256 value,
        bytes calldata data, bytes calldata operatorData) external;
    function controllerRedeem(address tokenHolder, uint256 value,
        bytes calldata data, bytes calldata operatorData) external;
}
```

## Transfer Hook Pattern

```solidity
interface ITransferRestrictions {
    function validateTransfer(
        bytes32 partition,
        address operator,
        address from,
        address to,
        uint256 value,
        bytes calldata data
    ) external view returns (bool valid, bytes1 escCode, string memory reason);
}

contract SecurityToken is ERC1400Base {
    ITransferRestrictions public restrictions;

    function _beforeTokenTransfer(
        bytes32 partition, address from, address to, uint256 value
    ) internal override {
        (bool valid, bytes1 code, string memory reason) =
            restrictions.validateTransfer(partition, msg.sender, from, to, value, "");
        require(valid, string(abi.encodePacked("ERC1400: ", reason)));
    }
}
```

## Lockup Migration (Reg D → Unrestricted)

```solidity
function convertLockedToUnrestricted(address holder) external onlyController {
    require(block.timestamp >= lockupExpiry[holder], "ERC1400: lockup not expired");
    uint256 lockedBal = balanceOfByPartition(PARTITION_LOCKED, holder);
    require(lockedBal > 0, "ERC1400: nothing to convert");

    _redeemByPartition(PARTITION_LOCKED, holder, lockedBal, "");
    _issueByPartition(PARTITION_UNRESTRICTED, holder, lockedBal, "");

    emit PartitionConverted(holder, PARTITION_LOCKED, PARTITION_UNRESTRICTED, lockedBal);
}
```
